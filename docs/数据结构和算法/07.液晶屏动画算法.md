### 丝滑移动位置
```C
/*
丝滑移动位置
a 当前值
a_trg 期待值
b 速度
*/
void ui_run(int* a, int* a_trg, int b){
	if(*a < *a_trg){ // 当前小于期待
		*a+=b;				 // 当前加一小步(b)
		if(*a > *a_trg){// 如果当前值大于期待值则等于期待值
			*a = *a_trg;
		}
	} else if(*a > *a_trg){ // 与第一步类似
		*a-=b;
		if(*a < *a_trg){
			*a = *a_trg;
		}
	} else {
		*a = *a_trg;
	}
}
```
